定位到了。index() 里 User::all() 之后在循环访问 $u->posts，触发 N+1 查询。改成预加载 + 分页：

public function index(Request $request) {
    $users = User::with('posts')->where('active', true)->orderByDesc('created_at')->paginate(20);
    return view('users.index', compact('users'));
}
